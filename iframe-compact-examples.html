⚡ JavaScript управление:

// Получение iframe
const iframe = document.getElementById('radioPlayer');

// Воспроизведение
iframe.contentWindow.postMessage({action: 'play'}, '*');

// Пауза
iframe.contentWindow.postMessage({action: 'pause'}, '*');

// Управление громкостью (0-1)
iframe.contentWindow.postMessage({action: 'setVolume', volume: 0.7}, '*');

🔧 Полный пример с управлением:

<!DOCTYPE html>
<html>
<head>
    <title>Мой сайт с радио</title>
</head>
<body>
    <h1>Добро пожаловать на мой сайт!</h1>
    
    <!-- Компактный плеер -->
    <iframe id="radioPlayer" 
            src="iframe-compact.html" 
            width="400" 
            height="80" 
            frameborder="0"></iframe>
    
    <!-- Кнопки управления -->
    <button onclick="playRadio()">▶️ Воспроизвести</button>
    <button onclick="pauseRadio()">⏸️ Пауза</button>
    <button onclick="setVolume(0.5)">🔉 Громкость 50%</button>
    
    <script>
        function playRadio() {
            document.getElementById('radioPlayer').contentWindow.postMessage({action: 'play'}, '*');
        }
        function pauseRadio() {
            document.getElementById('radioPlayer').contentWindow.postMessage({action: 'pause'}, '*');
        }
        function setVolume(volume) {
            document.getElementById('radioPlayer').contentWindow.postMessage({action: 'setVolume', volume: volume}, '*');
        }
    </script>
</body>
</html>
